00277         static void nonMaximaSuppression(const cv::Mat& im, vectorCandidate& candidates, const float overlap=0.0f) {
00278 
00279                 // create a scratch space that we can draw on
00280                 const size_t N = candidates.size();
00281                 const cv::Rect bounds = cv::Rect(0,0,0,0) + im.size();
00282                 cv::Mat scratch = cv::Mat::zeros(im.size(), CV_8U);
00283 
00284                 // the current insertion position in the vector
00285                 size_t keep = 0;
00286 
00287                 /* iterate through the boxes, checking:
00288                  * 1) has the area under the box been painted?
00289                  * 2) if so, is it under the threshold?
00290                  * 3) if so, keep this box and paint the area
00291                  * 4) repeat
00292                  */
00293                 for (size_t n = 0; n < N; ++n) {
00294                         cv::Rect box = candidates[n].boundingBox() & bounds;
00295                         cv::Scalar boxsum = sum(scratch(box));
00296                         if (boxsum[0] / box.area() > overlap) continue;
00297                         scratch(box) = 1;
00298                         candidates[keep] = candidates[n];
00299                         keep++;
00300                 }
00301 
00302                 // simply delete the trailing end of the candidates
00303                 candidates.resize(keep);
00304         }
