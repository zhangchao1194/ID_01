00313                         Eigen::aligned_allocator<pcl::PlanarRegion<PointType> > > regions;

00316         pcl::PointCloud<pcl::Label>::Ptr labels(new pcl::PointCloud<pcl::Label>());
00317         std::vector<pcl::PointIndices> label_indices;
00318         std::vector<pcl::PointIndices> boundary_indices;

00320         plane_segmentation.segmentAndRefine(regions, model_coefficients,
00321                         inlier_indices, labels, label_indices, boundary_indices);
